REMOTE = "http://192.168.100.6:8766/mcp/"
_session_id = None

def parse_sse(data: bytes):
    text = data.decode('utf-8', errors='replace')
    for block in text.split('\n\n'):
        for line in block.split('\n'):
            line = line.rstrip('\r')
            if line.startswith('data:'):
                payload = line[5:].strip()
                if payload and payload != '[DONE]':
                    try:
                        yield json.loads(payload)
                    except json.JSONDecodeError:
                        pass

def forward(msg):
    global _session_id
    headers = {
        'Content-Type': 'application/json',
        'Accept': 'application/json, text/event-stream',
    }
    if _session_id:
        headers['mcp-session-id'] = _session_id

    req = urllib.request.Request(REMOTE, json.dumps(msg).encode(), headers)
    try:
        with urllib.request.urlopen(req, timeout=60) as r:
            sid = r.headers.get('mcp-session-id')
            if sid:
                _session_id = sid
            body = r.read()
            ct = r.headers.get('Content-Type', '')
            if 'text/event-stream' in ct:
                return list(parse_sse(body)), None
            elif body.strip():
                return [json.loads(body)], None

                return [], None

        return [], {"code": -32603, "message": f"HTTP {e.code}: {e.reason}"}
